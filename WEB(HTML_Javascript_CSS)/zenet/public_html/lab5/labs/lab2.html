function getName()
{
var name="";

    aToz = new RegExp;
    aToz = /[A-z]/;
    
while(keepTrying == 0)
    {
    name = prompt("Please type your name.");
    
    if(name=="") // no input
        {alert("Input is Blank.\n Please try again.")}
    else if(aToz.test(name.charAt(0))== false) // First letter is not A-z
        {alert("First character MUST be alphabet.")}
    else if(invalid.test(name)==true) // there is any letter that is not A-z or ' or _

}
