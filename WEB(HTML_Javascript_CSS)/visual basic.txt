Dim con As New SqlClient.SqlConnection("Data Source=(local);Initial Catalog=Saleco;Integrated Security=True")

Dim cmd As New SqlCommand()
cmd.Connection = con

con.Open()

' CHECK WHETHER THE ENTERED CODE IS VALID
cmd.CommandText = "SELECT COUNT(*) FROM PRODUCT WHERE P_CODE='" & Me.pcode.Text & "'"
Dim count As Integer = Integer.Parse(cmd.ExecuteScalar())

Dim title = "Info"

If count > 0 Then

cmd.CommandText = "UPDATE PRODUCT SET P_QOH+=" & Me.qty.Text & " where P_CODE='" & Me.pcode.Text & "'"


If IsNumeric(Me.qty.Text) Then

cmd.ExecuteNonQuery()

Dim msg = "Product quantity in stock updated"
' Add the title to the display.
MsgBox(msg, , title)


Else
Dim msg = "quantity entered should be numeric"
' Add the title to the display.
MsgBox(msg, , title)

End If

Else
Dim msg = "Product code does not exist!"

End If
